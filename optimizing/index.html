使用自定义的懒加载方法

import React from 'react'
let asyncComp = (importComp) => {
  return class extends React.Component {
    constructor() {
      this.state = {
        comp: null,
      }
    }
    componentDidMount() {
      importComp().then((res) => {
        this.setState({ comp: res.default })
      })
    }
    render() {
      let C = this.state.comp
      return C ? <C {...this.props} /> : null
    }
  }
}
export default asyncComp
