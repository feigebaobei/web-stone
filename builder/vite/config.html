configuration

默认配置文件：vite.config.js。
它是被 node.js 使用。

export default defineConfig(({ command, mode, ssrBuild }) => {
  if (command === 'serve') {
    return {
      // dev 独有配置
    }
  } else {
    // command === 'build'
    return {
      // build 独有配置
    }
  }
})
// 异步配置 // js中总是优先异步。
export default defineConfig(async ({ command, mode }) => {
  const data = await asyncFunction()
  return {
    // vite 配置
  }
})
// 可以引用环境变量后使用。
import { defineConfig, loadEnv } from 'vite'
