function hoc (SubComp) {
    return {
        mounted() {
            clog('...')
        },
        props: SubComp.props,
        render(h) {
            let children = ...
            return h(SubComp, {
                on: this.$listeners,
                props: this.$props,
                atts: this.$attrs,

    }
}
