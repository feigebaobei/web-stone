自制状态管理

利用vue3中reactive做状态管理。 好像单例模式。

import {createApp, reactive} from 'vue'
const obj = reactive({
    msg: 'string'
})
const appA = createApp({
    data() {
        return obj
    }
}).mount('#a')
const appB = createApp({
    data() {
        return obj
    }
}).mount('#b')

import {reactive} from 'vue'
export let store = {
    debug: true,
    state: reactive({
        msg: 'str'
    }),
    setMsgAction(nv) {
        if (this.debug) {
            clog('...')
        }
        this.state.msg = nv
    }
    clearMsgAction() {
        if (this.debug) {
            clog('...')
        }
        this.state.msg = ''
    }
}
// 然后在组件中引入并使用。
