vuedraggable

overview

支持vue1/vue2的拖动、排序组件。 基于sortable.js

feature

  • 兼容vue2 / transition-group
  • 支持取消
  • feature2

install

npm i vuedraggable

usage

<draggable v-model="myArr" group="people" @start="drag=true" @end="drag=false">
    <div v-for="(item) in myArr" :key="item.id">str</div>
</draggable>
<draggable v-model="myArr">
    <transition-group>
        <div v-for="..."></div>
    </transition-group>
</draggable>

import draggable from 'vuedraggable'
exprot default {
    components: {
        draggable
    }
}
