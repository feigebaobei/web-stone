  }
  resize(n) {
    let t = this.size()
    if (t === this.arr.length) {
      // null
    } else if (n < t) {
      this.arr = this.arr.slice(0, n)
    } else {
      // t < n
      let t = [...this.arr]
      while (t.length < n) {
        t.push(null)
      }
      this.arr = t
    }
  }
  push_back(v) {
    this.arr.splice(0, 1)
    this.arr.push(v)
  }
  pop_back() {

  }
