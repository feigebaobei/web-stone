var arr = [1,2,3,4,5,6,7]
let f = (arr, v) => {
if (!arr.length) {
    return false
} else {
    let p = arr.length >> 1
    let t = v - arr[p]
    if (t === 0) {
    return true
    } else if (t > 0) {
    return f(arr.slice(p + 1), v)
    } else {
    return f(arr.slice(0, p), v)
    }
}
}
